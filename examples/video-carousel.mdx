# Video Carousel

Build a carousel for video content with accessibility features and optimal user experience. This example handles video playback, controls, and responsive behavior.

## Basic Video Carousel

A simple video carousel with navigation and proper video controls:

<CodeGroup>
```tsx Component
import { Carousel } from 'react-any-carousal';
import { useRef, useEffect } from 'react';

export default function VideoCarousel() {
  const videos = [
    {
      id: 1,
      src: '/videos/tutorial-1.mp4',
      poster: '/images/video-thumb-1.jpg',
      title: 'Getting Started Tutorial',
    },
    {
      id: 2,
      src: '/videos/tutorial-2.mp4',
      poster: '/images/video-thumb-2.jpg',
      title: 'Advanced Features',
    },
    {
      id: 3,
      src: '/videos/tutorial-3.mp4',
      poster: '/images/video-thumb-3.jpg',
      title: 'Best Practices',
    },
  ];

  return (
    <Carousel
      scrollSnapType="start"
      scrollOffset={640}
      duration={600}
    >
      {videos.map((video) => (
        <div key={video.id} className="video-slide">
          <video
            controls
            poster={video.poster}
            preload="metadata"
            aria-label={video.title}
          >
            <source src={video.src} type="video/mp4" />
            <track
              kind="captions"
              src={`/captions/${video.id}.vtt`}
              srcLang="en"
              label="English"
            />
            Your browser does not support the video tag.
          </video>
          <div className="video-title">
            <h3>{video.title}</h3>
          </div>
        </div>
      ))}
    </Carousel>
  );
}
```

```css Styles

  margin-right: 24px;
  scroll-snap-align: start;
}

.video-slide video {
  width: 100%;
  height: 338px;
  background: #000;
  border-radius: 12px;
  display: block;
}

.video-title {
  margin-top: 12px;
  padding: 0 8px;
}

.video-title h3 {
  font-size: 1.1rem;
  font-weight: 600;
  color: #333;
  margin: 0;
}
```
</CodeGroup>

<Note>
**Important:** Set `scrollOffset` to match your video width (600px + margins) to ensure one video scrolls at a time.
</Note>

## Video Carousel with Playback Management

Automatically pause videos when they scroll out of view:

<CodeGroup>
```tsx Component
import { Carousel } from 'react-any-carousal';
import { useRef, useEffect, useState } from 'react';

export default function ManagedVideoCarousel() {
  const [activeVideo, setActiveVideo] = useState<number>(0);
  const videoRefs = useRef<(HTMLVideoElement | null)[]>([]);

  const videos = [
    { id: 1, src: '/videos/demo-1.mp4', poster: '/images/poster-1.jpg' },
    { id: 2, src: '/videos/demo-2.mp4', poster: '/images/poster-2.jpg' },
    { id: 3, src: '/videos/demo-3.mp4', poster: '/images/poster-3.jpg' },
    { id: 4, src: '/videos/demo-4.mp4', poster: '/images/poster-4.jpg' },
  ];

  // Pause all videos except the active one
  useEffect(() => {
    videoRefs.current.forEach((video, index) => {
      if (video && index !== activeVideo) {
        video.pause();
      }
    });
  }, [activeVideo]);

  const handleVideoInView = (index: number) => {
    setActiveVideo(index);
  };

  return (
    <div className="video-carousel-container">
      <Carousel
        scrollSnapType="center"
        scrollOffset={700}
        duration={700}
        scrollEasing="cubic-bezier(0.25, 1, 0.5, 1)"
      >
        {videos.map((video, index) => (
          <div
            key={video.id}
            className="managed-video-slide"
            onClick={() => handleVideoInView(index)}
          >
            <video
              ref={(el) => (videoRefs.current[index] = el)}
              controls
              poster={video.poster}
              preload="metadata"
              playsInline
            >
              <source src={video.src} type="video/mp4" />
              Your browser does not support the video tag.
            </video>
            <div className="video-indicator">
              {index + 1} / {videos.length}
            </div>
          </div>
        ))}
      </Carousel>
    </div>
  );
}
```

```css Styles

  width: 100%;
  height: 383px;
  border-radius: 16px;
  background: #000;
  box-shadow: 0 8px 24px rgba(0, 0, 0, 0.2);
}

  position: absolute;
  top: 16px;
  right: 16px;

```
</CodeGroup>

<Tip>
Use `playsInline` attribute to prevent videos from opening in fullscreen on mobile devices.
</Tip>

## Accessibility Considerations

<Steps>
  <Step title="Provide captions">
    Always include caption tracks for videos:
    
    ```tsx
    <track
      kind="captions"
      src="/captions/video.vtt"
      srcLang="en"
      label="English"
    />
    ```
  </Step>
  
  <Step title="Use aria-label">
    Add descriptive labels to video elements:
    
    ```tsx
    <video aria-label="Product demo video" controls>
      {/* ... */}
    </video>
    ```
  </Step>
  
  <Step title="Include keyboard navigation">
    The Carousel component automatically supports keyboard navigation. Ensure videos are keyboard accessible with native controls.
  </Step>
  
  <Step title="Provide alternatives">
    Always include the text "Your browser does not support the video tag" as fallback content.
  </Step>
</Steps>

## Video Carousel with Custom Controls

Create a custom play/pause overlay:

<CodeGroup>
```tsx Component
import { Carousel } from 'react-any-carousal';
import { useRef, useState } from 'react';

export default function CustomControlsVideoCarousel() {
  const videos = [
    { id: 1, src: '/videos/product-1.mp4', poster: '/images/product-1.jpg' },
    { id: 2, src: '/videos/product-2.mp4', poster: '/images/product-2.jpg' },
    { id: 3, src: '/videos/product-3.mp4', poster: '/images/product-3.jpg' },
  ];

  return (
    <Carousel
      scrollSnapType="start"
      scrollOffset={800}
      duration={500}
      theme="dark"
    >
      {videos.map((video) => (
        <VideoCard key={video.id} video={video} />
      ))}
    </Carousel>
  );
}

function VideoCard({ video }: { video: { id: number; src: string; poster: string } }) {
  const videoRef = useRef<HTMLVideoElement>(null);
  const [isPlaying, setIsPlaying] = useState(false);

  const togglePlay = () => {
    if (videoRef.current) {
      if (isPlaying) {
        videoRef.current.pause();
      } else {
        videoRef.current.play();
      }
      setIsPlaying(!isPlaying);
    }
  };

  return (
    <div className="custom-video-card">
      <video
        ref={videoRef}
        poster={video.poster}
        preload="metadata"
        playsInline
        onClick={togglePlay}
      >
        <source src={video.src} type="video/mp4" />
      </video>
      {!isPlaying && (
        <button
          className="play-button"
          onClick={togglePlay}
          aria-label="Play video"
        >
          <svg width="64" height="64" viewBox="0 0 64 64">
            <circle cx="32" cy="32" r="32" fill="rgba(255,255,255,0.9)" />
            <polygon points="26,20 26,44 44,32" fill="#333" />
          </svg>
        </button>
      )}
    </div>
  );
}
```

```css Styles

}

}
```
</CodeGroup>

## Performance Tips

<Steps>
  <Step title="Use preload='metadata'">
    Load only video metadata initially to reduce bandwidth:
    
    ```tsx
    <video preload="metadata" />
    ```
  </Step>
  
  <Step title="Provide poster images">
    Always include a poster attribute for better perceived performance:
    
    ```tsx
    <video poster="/images/thumbnail.jpg" />
    ```
  </Step>
  
  <Step title="Optimize video files">
    Use compressed formats (H.264 for broad compatibility) and appropriate resolutions (720p or 1080p).
  </Step>
  
  <Step title="Consider lazy loading">
    For long carousels, dynamically load video sources as they come into view.
  </Step>
</Steps>

## Prop Reference

| Prop | Value | Purpose |
|------|-------|----------|
| `scrollSnapType` | `"start"` or `"center"` | Alignment of videos after scrolling |
| `scrollOffset` | `640-800` | Match to video width for one-at-a-time scrolling |
| `duration` | `500-700` | Smooth transition between videos |
| `theme` | `"dark"` | Use dark theme for video carousels |
| `scrollEasing` | `"cubic-bezier(0.25, 1, 0.5, 1)"` | Smooth, natural scrolling |

<Note>
The `theme="dark"` prop changes navigation button colors to work better with video backgrounds.
</Note>

## Common Issues

### Videos not playing on mobile
- Add `playsInline` attribute to prevent fullscreen
- Require user interaction to start playback (browser policy)

### Choppy scrolling
- Reduce video dimensions or bitrate
- Use `preload="none"` for many videos
- Consider using poster images with play buttons

### Audio conflicts
- Pause other videos when one starts playing
- Use the playback management example above
