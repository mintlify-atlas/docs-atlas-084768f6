# Full-Page Carousel

Build stunning full-width carousels perfect for hero sections, landing pages, and immersive presentations. Each slide takes up the entire viewport width.

## Basic Full-Width Carousel

Create a full-width carousel where each slide spans the entire container:

<CodeGroup>
```tsx Component
import { Carousel } from 'react-any-carousal';
import { useEffect, useState } from 'react';

export default function FullPageCarousel() {
  const [containerWidth, setContainerWidth] = useState(1200);

  useEffect(() => {
    const updateWidth = () => {
      setContainerWidth(window.innerWidth);
    };
    
    updateWidth();
    window.addEventListener('resize', updateWidth);
    return () => window.removeEventListener('resize', updateWidth);
  }, []);

  const slides = [
    {
      id: 1,
      title: 'Welcome to Our Platform',
      subtitle: 'Build amazing experiences with ease',
      image: '/images/hero-1.jpg',
      cta: 'Get Started',
    },
    {
      id: 2,
      title: 'Powerful Features',
      subtitle: 'Everything you need in one place',
      image: '/images/hero-2.jpg',
      cta: 'Learn More',
    },
    {
      id: 3,
      title: 'Join Thousands of Users',
      subtitle: 'Start your journey today',
      image: '/images/hero-3.jpg',
      cta: 'Sign Up Free',
    },
  ];

  return (
    <div className="full-page-carousel-wrapper">
      <Carousel
        scrollSnapType="start"
        scrollOffset={containerWidth}
        duration={800}
        autoSlideInterval={6000}
        scrollEasing="cubic-bezier(0.25, 1, 0.5, 1)"
        theme="dark"
      >
        {slides.map((slide) => (
          <div key={slide.id} className="full-page-slide">
            <div className="slide-background">
              <img src={slide.image} alt={slide.title} />
              <div className="slide-overlay" />
            </div>
            <div className="slide-content">
              <h1 className="slide-title">{slide.title}</h1>
              <p className="slide-subtitle">{slide.subtitle}</p>
              <button className="slide-cta">{slide.cta}</button>
            </div>
          </div>
        ))}
      </Carousel>
    </div>
  );
}
```

## Key Configuration for Full-Page

<Steps>
  <Step title="Set scrollOffset to container width">
    Use dynamic width calculation:
    
    ```tsx
    const [width, setWidth] = useState(window.innerWidth);
    
    useEffect(() => {
      const handleResize = () => setWidth(window.innerWidth);
      window.addEventListener('resize', handleResize);
      return () => window.removeEventListener('resize', handleResize);
    }, []);
    
    <Carousel scrollOffset={width}>
    ```
  </Step>
  
  <Step title="Use scrollSnapType='start'">
    Ensures slides align perfectly to the left edge:
    
    ```tsx
    scrollSnapType="start"
    ```
  </Step>
  
  <Step title="Set slides to 100vw">
    Each slide should span the full viewport:
    
    ```css
    .slide {
      min-width: 100vw;
      scroll-snap-align: start;
    }
    ```
  </Step>
  
  <Step title="Increase duration for smooth transitions">
    Full-page slides benefit from slightly longer transitions:
    
    ```tsx
    duration={800} // 800-1000ms recommended
    ```
  </Step>
</Steps>

## Vertical Full-Page Carousel

While the Carousel component is designed for horizontal scrolling, you can create a vertical-style experience:

```tsx
import { Carousel } from 'react-any-carousal';

// Note: This still scrolls horizontally, but mimics vertical layout
export default function VerticalStyleCarousel() {
  const [containerWidth, setContainerWidth] = useState(window.innerWidth);

  useEffect(() => {
    const updateWidth = () => setContainerWidth(window.innerWidth);
    updateWidth();
    window.addEventListener('resize', updateWidth);
    return () => window.removeEventListener('resize', updateWidth);
  }, []);

  return (
    <div style={{ height: '100vh' }}>
      <Carousel
        scrollSnapType="start"
        scrollOffset={containerWidth}
        duration={1000}
        scrollEasing="cubic-bezier(0.25, 1, 0.5, 1)"
      >
        <Section1 />
        <Section2 />
        <Section3 />
      </Carousel>
    </div>
  );
}

function Section1() {
  return (
    <div className="vertical-section" style={{ background: '#1f2937' }}>
      <h1>Section 1</h1>
    </div>
  );
}

// Styles
.vertical-section {
  min-width: 100vw;
  height: 100vh;
  scroll-snap-align: start;
  display: flex;
  align-items: center;
  justify-content: center;
  color: white;
}
```

## Navigation Indicators

Add dots or progress indicators to show current slide:

```tsx
import { Carousel } from 'react-any-carousal';
import { useState, useRef, useEffect } from 'react';

export default function CarouselWithIndicators() {
  const [currentSlide, setCurrentSlide] = useState(0);
  const containerRef = useRef<HTMLDivElement>(null);
  const totalSlides = 3;

  const handleScroll = () => {
    if (containerRef.current) {
      const scrollLeft = containerRef.current.scrollLeft;
      const slideWidth = window.innerWidth;
      const index = Math.round(scrollLeft / slideWidth);
      setCurrentSlide(index);
    }
  };

  return (
    <div ref={containerRef} onScroll={handleScroll}>
      <Carousel
        scrollSnapType="start"
        scrollOffset={window.innerWidth}
        duration={800}
      >
        {/* Slides */}
      </Carousel>
      
      <div className="slide-indicators">
        {Array.from({ length: totalSlides }).map((_, index) => (
          <button
            key={index}
            className={`indicator ${index === currentSlide ? 'active' : ''}`}
            aria-label={`Go to slide ${index + 1}`}
          />
        ))}
      </div>
    </div>
  );
}
```

## Performance Optimization

<Steps>
  <Step title="Lazy load images">
    Only load images as they come into view:
    
    ```tsx
    <img src={image} alt={alt} loading="lazy" />
    ```
  </Step>
  
  <Step title="Use CSS transforms for animations">
    Prefer `transform` and `opacity` for smooth 60fps animations.
  </Step>
  
  <Step title="Optimize image sizes">
    Use appropriately sized images for different screen sizes:
    
    ```tsx
    <img
      srcSet="hero-small.jpg 640w, hero-large.jpg 1920w"
      sizes="100vw"
    />
    ```
  </Step>
  
  <Step title="Debounce resize listeners">
    Prevent excessive recalculations on window resize.
  </Step>
</Steps>

## Accessibility Best Practices

<Note>
The Carousel component includes built-in ARIA attributes (`role="region"`, `aria-label="Carousel"`) and keyboard navigation support.
</Note>

### Additional Considerations

- **Pause auto-slide on focus:** Auto-sliding automatically stops when users interact with navigation
- **Provide text alternatives:** Ensure all visual content has text descriptions
- **Sufficient color contrast:** Maintain WCAG AA standards for text over images
- **Avoid motion sickness:** Keep `autoSlideInterval` at 5000ms or higher

## Prop Reference for Full-Page Carousels

| Prop | Value | Purpose |
|------|-------|----------|
| `scrollSnapType` | `"start"` | Align slides to viewport edge |
| `scrollOffset` | `window.innerWidth` | Scroll exactly one viewport width |
| `duration` | `800-1000` | Longer for smoother full-page transitions |
| `autoSlideInterval` | `6000-8000` | Give users time to read content |
| `scrollEasing` | `"cubic-bezier(0.25, 1, 0.5, 1)"` | Smooth, natural easing |
| `theme` | `"dark"` | Better for full-page overlays |

## Common Issues

### Slides not aligning properly
- Ensure each slide has `min-width: 100vw`
- Verify `scrollOffset` matches container width
- Check that `scroll-snap-align: start` is applied

### Jumpy scrolling on mobile
- Use `touch-action: pan-x` on the carousel container
- Avoid conflicting scroll handlers

### Auto-slide timing issues
- Increase `autoSlideInterval` to 6000ms or more
- Ensure `duration` is less than interval
