# Custom Components in Carousel

The Carousel component accepts any React elements as children, making it perfect for complex UI components like product cards, testimonials, blog posts, and more.

## Product Card Carousel

Create a beautiful e-commerce product carousel:

<CodeGroup>
```tsx Component
import { Carousel } from 'react-any-carousal';

interface Product {
  id: number;
  name: string;
  price: number;
  image: string;
  rating: number;
  inStock: boolean;
}

export default function ProductCarousel() {
  const products: Product[] = [
    {
      id: 1,
      name: 'Wireless Headphones',
      price: 99.99,
      image: '/images/headphones.jpg',
      rating: 4.5,
      inStock: true,
    },
    {
      id: 2,
      name: 'Smart Watch Pro',
      price: 299.99,
      image: '/images/watch.jpg',
      rating: 4.8,
      inStock: true,
    },
    {
      id: 3,
      name: 'Laptop Stand',
      price: 49.99,
      image: '/images/stand.jpg',
      rating: 4.3,
      inStock: false,
    },
    {
      id: 4,
      name: 'Mechanical Keyboard',
      price: 149.99,
      image: '/images/keyboard.jpg',
      rating: 4.7,
      inStock: true,
    },
  ];

  return (
    <Carousel
      scrollSnapType="start"
      scrollOffset={320}
      duration={600}
      theme="light"
    >
      {products.map((product) => (
        <ProductCard key={product.id} product={product} />
      ))}
    </Carousel>
  );
}

function ProductCard({ product }: { product: Product }) {
  return (
    <div className="product-card">
      <div className="product-image-container">
        <img src={product.image} alt={product.name} />
        {!product.inStock && (
          <div className="out-of-stock-badge">Out of Stock</div>
        )}
      </div>
      <div className="product-info">
        <h3 className="product-name">{product.name}</h3>
        <div className="product-rating">
          {'★'.repeat(Math.floor(product.rating))}
          {'☆'.repeat(5 - Math.floor(product.rating))}
          <span className="rating-value">{product.rating}</span>
        </div>
        <div className="product-footer">
          <span className="product-price">${product.price}</span>
          <button
            className="add-to-cart-btn"
            disabled={!product.inStock}
          >
            {product.inStock ? 'Add to Cart' : 'Sold Out'}
          </button>
        </div>
      </div>
    </div>
  );
}
```

      author: 'Sam Patel',

      category: 'Performance',
      image: '/images/blog-3.jpg',

  return (
    <Carousel
      scrollSnapType="start"
      scrollOffset={380}

      theme="light"

      {posts.map((post) => (
        <BlogCard key={post.id} post={post} />
      ))}

function BlogCard({ post }: { post: BlogPost }) {
  const formatDate = (dateString: string) => {
    return new Date(dateString).toLocaleDateString('en-US', {
      month: 'short',
      day: 'numeric',
      year: 'numeric',
    });
  };

  return (
    <article className="blog-card">
      <div className="blog-image">
        <img src={post.image} alt={post.title} />
        <div className="blog-category">{post.category}</div>
      </div>
      <div className="blog-content">
        <h3 className="blog-title">{post.title}</h3>
        <p className="blog-excerpt">{post.excerpt}</p>
        <div className="blog-meta">
          <div className="blog-author">
            <span className="author-name">{post.author}</span>
            <span className="blog-date">{formatDate(post.date)}</span>
          </div>
          <span className="read-time">{post.readTime}</span>
        </div>
        <button className="read-more-btn">Read Article →</button>
      </div>
    </article>
  );
}

## Key Best Practices

<Steps>
  <Step title="Match scrollOffset to component width">
    Calculate the total width including margins:
    
    ```tsx
    // For 360px card + 20px margin:
    scrollOffset={380}
    ```
  </Step>
  
  <Step title="Add scroll-snap-align to each card">
    Ensure smooth snapping behavior:
    
    ```css
    .your-card {
      scroll-snap-align: start; /* or center */
    }
    ```
  </Step>
  
  <Step title="Use semantic HTML">
    Choose appropriate elements (`<article>`, `<section>`) for better accessibility.
  </Step>
  
  <Step title="Handle interactive elements">
    Buttons and links inside carousel items work normally. No special handling needed.
  </Step>
</Steps>

## Responsive Component Carousel

Make your carousel components responsive:

```tsx
import { Carousel } from 'react-any-carousal';
import { useState, useEffect } from 'react';

export default function ResponsiveCarousel() {
  const [scrollOffset, setScrollOffset] = useState(380);

  useEffect(() => {
    const updateOffset = () => {
      if (window.innerWidth < 640) {
        setScrollOffset(280); // Smaller cards on mobile
      } else if (window.innerWidth < 1024) {
        setScrollOffset(340); // Medium cards on tablet
      } else {
        setScrollOffset(380); // Full-size cards on desktop
      }
    };

    updateOffset();
    window.addEventListener('resize', updateOffset);
    return () => window.removeEventListener('resize', updateOffset);
  }, []);

  return (
    <Carousel
      scrollSnapType="start"
      scrollOffset={scrollOffset}
      duration={600}
    >
      {/* Your components */}
    </Carousel>
  );
}
```

<Tip>
**Pro tip:** Use CSS custom properties to sync component width with JavaScript:

```tsx
<div style={{ '--card-width': `${cardWidth}px` } as React.CSSProperties}>
  <Carousel scrollOffset={cardWidth + 20}>
    {/* Cards use var(--card-width) in CSS */}
  </Carousel>
</div>
```
</Tip>

## Prop Reference for Component Carousels

| Prop | Recommended Value | Purpose |
|------|-------------------|----------|
| `scrollSnapType` | `"start"` | Align cards to the start (left edge) |
| `scrollOffset` | Component width + margins | Scroll one component at a time |
| `duration` | `600-700` | Smooth, not too fast transitions |
| `theme` | `"light"` | Works well with card-based designs |
| `autoSlideInterval` | `5000+` | Optional, 5+ seconds for reading content |

## Common Patterns

### Cards with Actions
Buttons and links inside cards work normally. Click events are not affected by carousel scrolling.

### Loading States
Show skeleton cards while data loads:

```tsx
<Carousel scrollSnapType="start" scrollOffset={320}>
  {loading
    ? Array(3).fill(0).map((_, i) => <SkeletonCard key={i} />)
    : data.map((item) => <Card key={item.id} data={item} />)
  }
</Carousel>
```

### Empty States
Handle empty data gracefully:

```tsx
{items.length > 0 ? (
  <Carousel>{/* items */}</Carousel>
) : (
  <div className="empty-state">No items to display</div>
)}
```
